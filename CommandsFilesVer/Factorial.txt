in ; Указать факториал какого числа хочешь получить
pop ax
push 1
pop bx
push bx ; Пушим в стэк 1, чтобы умножение началось с единички

call as:

out ; Выводим результат на экран
hlt

;--------------------------------------------------------------------------
as:
    push ax ; Пушим число которое указали в начале и то, до которого умножение добралось в данный момент.
    push bx

    jb done: ; Сравниваем эти два числа, если число не достигло необходимого продолжаем

    push bx+1

    push bx+1 ; Увеличиваем значение числа для сравнения
    pop bx

    call as: ; Вызываем функцию снова

    mul ; Рекурсивно перемножаем все лежащие в стэке значения

    done:
ret
