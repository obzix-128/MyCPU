in
in ; Координаты центра

in ; Радиус

pop cx

pop bx
pop ax ; Кладём полученные значения в регистры

push 0
push 0

pop dx
pop ex

push 0
pop hx ; Создаю счётчик в регистре hx

jmp xy:

xy:
    push 40
    push dx
    jb y: ; Пока x не достигнет 9 буду увеличивать его на 1 и класть вместе с текущим значением y

    push dx ; Вычисляю значение первой скобки в уравнении окружности
    push ax
    sub
    sqr

    push ex ; Вычисляю значение второй скобки в уравнении окружности
    push bx
    sub
    sqr

    add ; Складываю значение скобок

    push cx
    sub

    push 0

    jb check: ; Если результат вычитания скобок и радиуса меньше нуля, в память пойдёт единичка, если нет, то нолик

    push 0
    pop [hx]

    push hx+1
    pop hx

    jmp up: ; Положил в память нолик, отправляюсь в начало цикла  но перед этим меняю координаты точки

    check:

    push 1
    pop [hx]

    push hx+1
    pop hx

    jmp up: ; Положил в память единичку, отправляюсь в начало цикла, но перед этим меняю координаты точки

    up:

    push dx+1 ; Увеличил x на 1
    pop dx

    jmp xy: ; Повторить цикл

    y:
        push 39
        push ex
        jb done: ; Пока y не достигнет 9, буду повторять

        push ex+1
        pop ex

        push 0
        pop dx

        jmp xy: ; Вернуться в начало с увеличенным y

done:
